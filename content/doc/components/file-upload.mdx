# File Upload

A modern file upload component that supports drag and drop, file type validation, and more.

## Installation

```bash
npm install @voidui/file-upload
# or
yarn add @voidui/file-upload
```

## Usage

```tsx
import { FileUpload } from "@voidui/file-upload"

export function UploadExample() {
  return (
    <FileUpload
      maxFiles={5}
      maxSize={5 * 1024 * 1024} // 5MB
      accept={{
        'image/*': ['.png', '.jpg', '.jpeg', '.gif'],
        'application/pdf': ['.pdf']
      }}
      onDrop={(acceptedFiles) => {
        console.log(acceptedFiles)
      }}
    >
      {({ getRootProps, getInputProps }) => (
        <div {...getRootProps()}>
          <input {...getInputProps()} />
          <p>Drag and drop files here, or click to select files</p>
        </div>
      )}
    </FileUpload>
  )
}
```

## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| maxFiles | number | 1 | Maximum number of files to accept |
| maxSize | number | Infinity | Maximum file size in bytes |
| accept | Record<string, string[]> | {} | Object of MIME types and their extensions |
| onDrop | (files: File[]) => void | - | Callback when files are dropped/selected |
| disabled | boolean | false | Disable the file upload |
| className | string | '' | Additional CSS class name |

## Examples

### Multiple file upload

```tsx
<FileUpload
  maxFiles={10}
  onDrop={(files) => console.log(files)}
>
  {({ getRootProps, getInputProps }) => (
    <div {...getRootProps()}>
      <input {...getInputProps()} />
      <p>Upload up to 10 files</p>
    </div>
  )}
</FileUpload>
```

### Custom dropzone

```tsx
<FileUpload onDrop={(files) => console.log(files)}>
  {({ getRootProps, getInputProps, isDragActive }) => (
    <div 
      {...getRootProps()}
      className={`p-8 border-2 border-dashed rounded-lg ${
        isDragActive ? 'border-blue-500' : 'border-gray-300'
      }`}
    >
      <input {...getInputProps()} />
      {isDragActive ? (
        <p>Drop the files here ...</p>
      ) : (
        <p>Drag and drop some files here, or click to select files</p>
      )}
    </div>
  )}
</FileUpload>
```
